*** calc() 계산 결과 양식 ***
담을 정보.
레벨링에 소모되는 자원
정예화에 소모되는 자원
스킬작에 소모되는 자원
소모되는 자원의 총합

소모되는 자원의 저장 방식
[자원ID, 수량], ...

즉, 아래와 같은 형식으로 결과 데이터를 저장할 것임
"result": {
	"opLeveling": {
		"exp": 필요경험치,
		"lmd": 필요용문폐
	},
	
	"elitePromotion": {
		"IDs": [자원ID1, 자원ID2, ...]
		"자원 ID": 수량,
		...
	]},
	
	"skillLeveling": {
		"IDs": [자원ID1, 자원ID2, ..
		"자원 ID": 수량,
		...
	},
	
	// 사실 꼭 필요한 건 아니지만, 나중에 결과 데이터 활용할 때 편하라고.
	"total": {
		"IDs": [자원ID1, 자원ID2, ..
		"자원 ID": 수량,
		...
	}
}
